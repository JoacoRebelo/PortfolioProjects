SELECT *
FROM international_debt
LIMIT 10;


SELECT 
    COUNT(DISTINCT country_name) AS total_distinct_countries
FROM international_debt;


--Para comprender las áreas en las que un país puede estar endeudado selecciono todas las categorías de deudas que hay.
SELECT
    DISTINCT indicator_code AS distinct_debt_indicators
FROM international_debt
ORDER BY distinct_debt_indicators;


--Busco el monto total de deuda (en USD) que deben los países.
SELECT 
    ROUND(SUM(debt)/1000000, 2) AS total_debt
FROM international_debt;


--Busco el país con mayor cantidad de deuda y muestro de cuánto es.
SELECT country_name, SUM(debt) AS total_debt
FROM international_debt
GROUP BY country_name
ORDER BY total_debt DESC
LIMIT 1;


--Miro el promedio de deuda en cada categoría.
SELECT indicator_code AS debt_indicator, indicator_name, AVG(debt) AS average_debt
FROM international_debt
GROUP BY debt_indicator, indicator_name
ORDER BY average_debt DESC
LIMIT 10;


--Averiguo qué país tiene la mayor cantidad de deuda en la categoría de deudas a largo plazo.
SELECT country_name, indicator_name
FROM international_debt
WHERE debt = (SELECT 
                 MAX(debt)
             FROM international_debt
             WHERE indicator_code = 'DT.AMT.DLXF.CD');


--Quiero saber si la deuda a largo plazo es el idnicador más común.
SELECT indicator_code, COUNT(indicator_code) AS indicator_count
FROM international_debt
GROUP BY indicator_code
ORDER BY indicator_count DESC, indicator_code DESC
LIMIT 20;


--Miro la cantidad máxima de deuda que tiene cada país.
SELECT country_name, MAX(debt) AS maximum_debt
FROM international_debt
GROUP BY country_name
ORDER BY maximum_debt DESC
LIMIT 10;
